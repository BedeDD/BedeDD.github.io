# Global WebMock configuration
            # - allow localhost connections
            # - googlechromelabs.github.io is used to get the latest versions for testing with webdrivers used in system tests
            # - storage.googleapis.com is used to download the latest versions of the chrome webdrivers
            # - 127.0.0.1 which is used by selenium we MUST provide it
            #             although we allow localhost connections
            allowed_connections = %w[
            googlechromelabs.github.io
            storage.googleapis.com
            127.0.0.1
            localhost:9200
            localhost:9200/_bulk
            ]
            WebMock.disable_net_connect!
            WebMock.allow_net_connect!(
              allow_localhost: true,
              allow: allowed_connections
            )
